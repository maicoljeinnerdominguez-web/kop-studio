    43→/* ─── Animated Counter ─── */
    44→function AnimatedCounter({ value, prefix = '', suffix = '' }: { value: number; prefix?: string; suffix?: string }) {
    45→  const [count, setCount] = useState(0);
    46→  useEffect(() => {
    47→    const duration = 1000;
    48→    const steps = 30;
    49→    const increment = value / steps;
    50→    let current = 0;
    51→    const timer = setInterval(() => {
    52→      current += increment;
    53→      if (current >= value) {
    54→        setCount(value);
    55→        clearInterval(timer);
    56→      } else {
    57→        setCount(Math.floor(current));
    58→      }
    59→    }, duration / steps);
    60→    return () => clearInterval(timer);
    61→  }, [value]);
    62→
    63→  return <span>{prefix}{count.toLocaleString('es-CO')}{suffix}</span>;
    64→}

   127→/* ─── OrderStatusSelect ─── */
   128→function OrderStatusSelect({
   129→  orderId,
   130→  currentStatus,
   131→  onStatusChange,
   132→}: {
   133→  orderId: string;
   134→  currentStatus: string;
   135→  onStatusChange: (orderId: string, newStatus: string) => void;
   136→}) {
   137→  const [isUpdating, setIsUpdating] = useState(false);
   138→  const status = currentStatus as OrderStatus;
   139→
   140→  const handleChange = async (newStatus: string) => {
   141→    if (newStatus === currentStatus) return;
   142→
   143→    setIsUpdating(true);
   144→    try {
   145→      const res = await fetch(`/api/orders/${orderId}`, {
   146→        method: 'PATCH',
   147→        headers: { 'Content-Type': 'application/json' },
   148→        body: JSON.stringify({ status: newStatus }),
   149→      });
   150→
   151→      if (!res.ok) {
   152→        const err = await res.json();
   153→        throw new Error(err.error || 'Error al actualizar');
   154→      }
   155→
   156→      const updated = await res.json();
   157→      onStatusChange(orderId, updated.status);
   158→      toast.success(`Estado actualizado a ${STATUS_LABELS[newStatus as OrderStatus]}`);
   159→    } catch {
   160→      toast.error('Error al actualizar el estado');
   161→    } finally {
   162→      setIsUpdating(false);
   163→    }
   164→  };
   165→
   166→  const colorClasses = STATUS_COLORS[status] || STATUS_COLORS.PENDING;
   167→
   168→  return (
   169→    <div className="relative inline-flex">
   170→      {isUpdating && (
   171→        <div className="absolute inset-0 z-10 flex items-center justify-center bg-[#0a0a0a]/80 rounded-sm">
   172→          <Loader2 className="w-3.5 h-3.5 animate-spin text-neutral-400" />
   173→        </div>
   174→      )}
   175→      <Select value={status} onValueChange={handleChange}>
   176→        <SelectTrigger
   177→          className={`inline-flex items-center gap-0 border ${colorClasses} hover:opacity-80 focus-visible:ring-1 focus-visible:ring-neutral-500 h-6 px-2 py-0 text-[10px] uppercase tracking-wider font-bold rounded-none bg-transparent cursor-pointer shadow-none transition-opacity`}
   178→        >
   179→          <SelectValue />
   180→        </SelectTrigger>
   181→        <SelectContent
   182→          position="popper"
   183→          className="bg-[#0a0a0a] border-[#1a1a1a] rounded-none"
   184→        >
   185→          {ALL_STATUSES.map((s) => (
   186→            <SelectItem
   187→              key={s}
   188→              value={s}
   189→              className={`text-xs uppercase tracking-wider font-medium rounded-none focus:bg-[#1a1a1a] ${
   190→                STATUS_COLORS[s]
   191→              } cursor-pointer`}
   192→            >
   193→              <span className="flex items-center gap-2">
   194→                <span
   195→                  className="w-2 h-2 rounded-full"
   196→                  style={{
   197→                    backgroundColor:
   198→                      s === 'PENDING'
   199→                        ? '#eab308'
   200→                        : s === 'PAID'
   201→                        ? '#10b981'
   202→                        : s === 'SHIPPED'
   203→                        ? '#f97316'
   204→                        : s === 'DELIVERED'
   205→                        ? '#22c55e'
   206→                        : '#ef4444',
   207→                  }}
   208→                />
   209→                {STATUS_LABELS[s]}
   210→              </span>
   211→            </SelectItem>
   212→          ))}
   213→        </SelectContent>
   214→      </Select>
   215→    </div>
   216→  );
   217→}

   299→    setLoading(true);

   301→      const res = await fetch('/api/admin');
   302→      if (!res.ok) throw new Error('Failed to fetch');
   303→      const data = await res.json();
   304→      setStats({
   305→        totalSales: data.totalSales,
   306→        pendingOrders: data.pendingOrders,
   307→        activeProducts: data.activeProducts,
   308→        totalOrders: data.totalOrders,
   309→      });
