   126тЖТ   126тЖТfunction ImageZoom({
   127тЖТ   127тЖТ  src,
   128тЖТ   128тЖТ  alt,
   129тЖТ   129тЖТ  isSelected,
   130тЖТ   130тЖТ}: {
   131тЖТ   131тЖТ  src: string
   132тЖТ   132тЖТ  alt: string
   133тЖТ   133тЖТ  isSelected: boolean
   134тЖТ   134тЖТ}) {
   135тЖТ   135тЖТ  const containerRef = useRef<HTMLDivElement>(null)
   136тЖТ   136тЖТ  const [isZoomed, setIsZoomed] = useState(false)
   137тЖТ   137тЖТ  const [zoomPosition, setZoomPosition] = useState({ x: 50, y: 50 })
   138тЖТ   138тЖТ
   139тЖТ   139тЖТ  const handleMouseMove = useCallback(
   140тЖТ   140тЖТ    (e: React.MouseEvent<HTMLDivElement>) => {
   141тЖТ   141тЖТ      if (!containerRef.current) return
   142тЖТ   142тЖТ      const rect = containerRef.current.getBoundingClientRect()
   143тЖТ   143тЖТ      const x = ((e.clientX - rect.left) / rect.width) * 100
   144тЖТ   144тЖТ      const y = ((e.clientY - rect.top) / rect.height) * 100
   145тЖТ   145тЖТ      setZoomPosition({ x, y })
   146тЖТ   146тЖТ    },
   147тЖТ   147тЖТ    []
   148тЖТ   148тЖТ  )
   149тЖТ   149тЖТ
   150тЖТ   150тЖТ  const handleMouseEnter = useCallback(() => setIsZoomed(true), [])
   151тЖТ   151тЖТ  const handleMouseLeave = useCallback(() => setIsZoomed(false), [])
   152тЖТ   152тЖТ
   153тЖТ   153тЖТ  if (!isSelected) return null
   154тЖТ   154тЖТ
   155тЖТ   155тЖТ  return (
   156тЖТ   156тЖТ    <div
   157тЖТ   157тЖТ      ref={containerRef}
   158тЖТ   158тЖТ      className="w-full h-full overflow-hidden cursor-zoom-in"
   159тЖТ   159тЖТ      onMouseMove={handleMouseMove}
   160тЖТ   160тЖТ      onMouseEnter={handleMouseEnter}
   161тЖТ   161тЖТ      onMouseLeave={handleMouseLeave}
   162тЖТ   162тЖТ    >
   163тЖТ   163тЖТ      <motion.img
   164тЖТ   164тЖТ        src={src}
   165тЖТ   165тЖТ        alt={alt}
   166тЖТ   166тЖТ        initial={{ opacity: 0 }}
   167тЖТ   167тЖТ        animate={{ opacity: 1 }}
   168тЖТ   168тЖТ        transition={{ duration: 0.3 }}
   169тЖТ   169тЖТ        className="w-full h-full object-cover transition-transform duration-200 ease-out img-blur-up"
   170тЖТ   170тЖТ        style={
   171тЖТ   171тЖТ          isZoomed
   172тЖТ   172тЖТ            ? {
   173тЖТ   173тЖТ                transform: 'scale(1.5)',
   174тЖТ   174тЖТ                transformOrigin: `${zoomPosition.x}% ${zoomPosition.y}%`,
   175тЖТ   175тЖТ              }
